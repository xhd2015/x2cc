458  inline DefaultLexcialToGrammarStream::DefaultLexcialToGrammarStream(
459  const StreamConvertor::WordStream& wdstream, const Gramma& g)
460  {

463  if(p.second==LexicalParser::TYPE_ID)
464  {
465  if(ID_IS_VALUE.find(p.first)!=ID_IS_VALUE.end())
466  {
467  this->push_back(g.gsyms.get(p.first));
468  }else{
469  this->push_back(g.gsyms.get("id"));
470  }
471  }else if(p.second==LexicalParser::TYPE_SINGLE){
472  this->push_back(g.gsyms.get(p.first));
473  }else if(p.second==LexicalParser::TYPE_NUMBER_BIN ||
474  p.second==LexicalParser::TYPE_NUMBER_HEX ||
475  p.second==LexicalParser::TYPE_NUMBER_DECIMAL){
476  this->push_back(g.gsyms.get("number"));
477  }else if(p.second == LexicalParser::TYPE_STRING){
478  this->push_back(g.gsyms.get("stringval"));
479  }else if(p.second == LexicalParser::TYPE_CHAR){
480  this->push_back(g.gsyms.get("charval"));
481  }else if(p.second == LexicalParser::TYPE_NOTE){
482  //pass
483  }else if(p.second == LexicalParser::TYPE_END){
484  this->push_back(g.gsyms.get("$"));
485  }else{
486  std::cout << "ERROR :" << LexicalParser::humanInfo[p.second] << std::endl;
487  }
488  }
